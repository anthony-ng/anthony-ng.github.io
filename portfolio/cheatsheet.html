SQL SELECT STATEMENTS

SELECT * FROM t

SELECT c1,c2 FROM t

SELECT c1,c2 FROM t
WHERE conditions

SELECT c1,c2 FROM t
WHERE conditions
ORDER BY c1 ASC,c2 DESC

SELECT DISTICT c1,c2
FROM t

SELECT c1, aggregate(c2 * c3)
FROM t
GROUP BY c1

SELECT c1, aggregate(c2 * c3)
FROM t
GROUP BY c1
HAVING c1 > v1

SQL OPERATORS

SELECT * from t

WHERE c1 [NOT] BETWEEN v1 AND v2

SELECT * FROM t
WHERE c1 [NOT ] IN (v1,v1,...)

SELECT * FROM t
WHERE c1 > v1 AND c1 < v2

SELECT * FROM t
WHERE c1 < v1 AND c1 < v2

SELECT * FROM t
WHERE c1 = v1

SELECT * FROM t
WHERE c1 <> v1
